void print_sums(void)
{
  int sum;

  for (int j = 0; j != 10; j++)
  {
    sum = sum + j;
  }
  printf("Sum is %d\n", sum);
  printf("Resetting...\n");

  sum = 0;
  for (int j = 0; j != 10; j++)
  {
    sum = sum + j;
  }
  printf("Sum is now %d\n", sum);
}

{
